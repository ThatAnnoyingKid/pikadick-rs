7/// Get the active toolchain
8pub fn get_rustup_active_toolchain() -> anyhow::Result<String> {
9    let output = std::process::Command::new("rustup")
10        .args(["show", "active-toolchain"])
11        .output()
12        .context("failed to spawn rustup")?;
13    ensure!(output.status.success(), "rustup exit code was non-zero");
14
15    let stdout = std::str::from_utf8(&output.stdout).context("output was not utf8")?;
16    let mut iter = stdout.split(' ');
17
18    let toolchain = iter.next().context("missing toolchain")?;
19
20    Ok(toolchain.to_string())
21}

23/// Get targets installed by rustup
24pub fn get_rustup_installed_targets<'a>(
25    toolchain: impl Into<Option<&'a str>>,
26) -> anyhow::Result<Vec<String>> {
27    let toolchain = toolchain.into();
28
29    let mut cmd = std::process::Command::new("rustup");
30    cmd.args(["target", "list", "--installed"]);
31
32    if let Some(toolchain) = toolchain {
33        cmd.args(["--toolchain", toolchain]);
34    }
35
36    let output = cmd.output().context("failed to spawn rustup")?;
37
38    ensure!(output.status.success(), "rustup exit code was non-zero");
39
40    let stdout = std::str::from_utf8(&output.stdout).context("output was not utf8")?;
41    Ok(stdout
42        .split('\n')
43        .map(|target_str| target_str.trim())
44        .filter(|target_str| !target_str.is_empty())
45        .map(|target_str| target_str.to_string())
46        .collect())
47}

49/// Make a target triple into a deb arch
50pub fn make_deb_arch(target: &str) -> anyhow::Result<&'static str> {
51    let mut iter = target.split('-');
52    let arch = iter.next().context("missing arch string")?;
53    let vendor = iter.next().context("missing vendor string")?;
54    let os_type = iter.next().context("missing os type")?;
55    let environment_type = iter.next();
56
57    match (arch, vendor, os_type, environment_type) {
58        ("aarch64", _, _, _) => Ok("arm64"),
59        (arm, _, _, Some(gnueabi)) if arm.starts_with("arm") && gnueabi.ends_with("hf") => {
60            Ok("armhf")
61        }
62        _ => bail!("failed to convert target triple `{}` to a deb arch", target),
63    }
64}
