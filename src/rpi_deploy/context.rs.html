33impl Context {
34    /// Create the context.
35    pub fn new() -> anyhow::Result<Self> {
36        ssh2::init();
37
38        println!("Fetching cargo metadata...");
39        let metadata = cargo_metadata::MetadataCommand::new()
40            .exec()
41            .context("failed to get cargo metadata")?;
42
43        let root_package = metadata
44            .root_package()
45            .context("failed to locate root package")?
46            .clone();
47
48        println!("Loading config...");
49        let cargo_toml_config = CargoTomlConfig::load_from_package(&root_package)?;
50        let file_config = FileConfig::new()?;
51
52        let rustup_toolchain =
53            get_rustup_active_toolchain().context("failed to get the active toolchain")?;
54        let rustup_installed_targets = get_rustup_installed_targets(rustup_toolchain.as_str())
55            .with_context(|| {
56                format!("failed to get rustup toolchains for toolchain \"{rustup_toolchain}\"",)
57            })?;
58        println!("Current Toolchain: {rustup_toolchain}");
59        println!("Installed Targets: {rustup_installed_targets:#?}");
60
61        println!();
62
63        Ok(Self {
64            cargo_toml_config,
65            file_config,
66
67            // rustup_toolchain,
68            rustup_installed_targets,
69
70            metadata,
71            root_package,
72
73            cross_config: None,
74        })
75    }

77    /// Get the cargo deb package path
78    pub fn get_cargo_deb_package_path(&self, target: &str) -> anyhow::Result<Utf8PathBuf> {
79        let mut path = self.metadata.target_directory.clone();
80        let name = self.get_cargo_deb_package_name(target)?;
81
82        path.extend([target, "debian", &name]);
83
84        Ok(path)
85    }
86
87    /// Get the cargo deb package name
88    pub fn get_cargo_deb_package_name(&self, target: &str) -> anyhow::Result<String> {
89        let deb_arch = make_deb_arch(target)?;
90        let name = &self.root_package.name;
91        let version = &self.root_package.version;
92        let revision = &self
93            .root_package
94            .metadata
95            .as_object()
96            .and_then(|metadata| metadata.get("deb")?.as_object()?.get("revision")?.as_str())
97            .unwrap_or("1");
98
99        Ok(format!("{name}_{version}-{revision}_{deb_arch}.deb"))
100    }
101
102    /// Set the cross config path
103    pub fn set_cross_config<P>(&mut self, path: P)

105        P: AsRef<Utf8Path>,
106    {
107        self.cross_config = Some(path.as_ref().into());
108    }

110    /// Package a target
111    pub fn package_target(&self, target: &str) -> anyhow::Result<()> {
112        println!("Packaging for \"{target}\"...");
113        println!();
114
115        ensure!(
116            self.rustup_target_is_installed(target),
117            "rustup target \"{target}\" is not installed",
118        );
119
120        // Building
121        {
122            println!("Building...");
123            let mut command = std::process::Command::new("across");
124            command.args(["--target", target]).arg("--release");
125            if let Some(cross_config) = self.cross_config.as_ref() {
126                command.arg("--config").arg(cross_config);
127            }
128            let status = command.status().context("failed to spawn build command")?;
129            ensure!(status.success(), "build command failed");
130            println!();
131        }
132
133        // Packaging
134        {
135            println!("Packaging...");
136            let status = std::process::Command::new("cargo")
137                .arg("deb")
138                .args(["--target", target])
139                .arg("--no-build")
140                .arg("--no-strip")
141                .arg("-v")
142                .status()
143                .context("failed to spawn package command")?;
144            ensure!(status.success(), "build command failed");
145            println!();
146        }
147
148        println!("Done packaging \"{target}\"");
149        println!();
150
151        Ok(())
152    }
153
154    /// Returns true if a given rustup target is installed
155    pub fn rustup_target_is_installed(&self, target: &str) -> bool {
156        self.rustup_installed_targets
157            .iter()
158            .any(|installed_target| installed_target == target)
159    }
