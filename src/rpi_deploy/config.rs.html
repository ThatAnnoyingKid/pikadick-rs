4const CARGO_TOML_CONFIG_METADATA_KEY: &str = "rpi-deploy";
5const FILE_CONFIG_FILE_NAME: &str = "rpi-deploy.toml";

7/// Config for this tool in Cargo.toml
8#[derive(serde::Deserialize)]
9pub struct CargoTomlConfig {
10    /// Targets to build
11    pub targets: Vec<String>,
12}
13
14impl CargoTomlConfig {
15    /// Load this from a given cargo_metadata package
16    pub fn load_from_package(package: &cargo_metadata::Package) -> anyhow::Result<Self> {
17        let value = package
18            .metadata
19            .get(CARGO_TOML_CONFIG_METADATA_KEY)
20            .with_context(|| {
21                let package_name = package.name.as_str();
22                format!(
23                    "missing `{CARGO_TOML_CONFIG_METADATA_KEY}` key in metadata for `{package_name}`"
24                )
25            })?;
26
27        serde_json::from_value(value.clone())
28            .context("failed to parse `Cargo.toml` metadata config")
29    }
30}

32/// The config file
33#[derive(serde::Deserialize)]
34pub struct FileConfig {
35    /// Machine configs
36    #[serde(flatten)]
37    pub machines: HashMap<String, MachineConfig>,
38}
39
40impl FileConfig {
41    /// Load a file config.
42    pub fn new() -> anyhow::Result<Self> {
43        let config_str = match std::fs::read_to_string(FILE_CONFIG_FILE_NAME) {
44            Ok(config_str) => config_str,
45            Err(e) if e.kind() == std::io::ErrorKind::NotFound => return Ok(Self::empty()),
46            Err(e) => {
47                return Err(e).with_context(|| {
48                    format!("failed to read file config at `{}`", FILE_CONFIG_FILE_NAME)
49                })?;
50            }
51        };
52
53        toml::from_str(&config_str).context("failed to parse file config")
54    }
55
56    /// Make an empty config.
57    fn empty() -> Self {
58        Self {
59            machines: HashMap::new(),
60        }
61    }
62
63    /// Get the machine config for a key.
64    pub fn get_machine_config(&self, name: &str) -> Option<&MachineConfig> {
65        self.machines.get(name)
66    }
67}
