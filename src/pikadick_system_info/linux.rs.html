29/// A cache for system data
30#[derive(Debug)]
31pub(crate) struct CacheContext {
32    sysinfo: OnceCell<SysInfo>,
33    utsname: OnceCell<UtsName>,
34}
35
36impl CacheContext {
37    /// Make a new cache context
38    pub fn new() -> Self {
39        Self {
40            sysinfo: OnceCell::new(),
41            utsname: OnceCell::new(),
42        }
43    }
44
45    /// Get the system info
46    fn get_sysinfo(&self) -> Result<&SysInfo, Error> {
47        self.sysinfo
48            .get_or_try_init(|| sysinfo().map_err(|error| Error::Io(std::io::Error::from(error))))
49    }
50
51    /// Get the utsname
52    fn get_utsname(&self) -> Result<&UtsName, Error> {
53        self.utsname
54            .get_or_try_init(|| uname().map_err(|error| Error::Io(std::io::Error::from(error))))
55    }
56
57    /// Get the time the system was booted
58    pub fn get_boot_time(&self) -> Result<SystemTime, Error> {
59        Ok(SystemTime::now() - self.get_uptime()?)
60    }
61
62    /// Get the uptime
63    pub fn get_uptime(&self) -> Result<Duration, Error> {
64        Ok(self.get_sysinfo()?.uptime())
65    }
66
67    /// Get the hostname
68    pub fn get_hostname(&self) -> Result<String, Error> {
69        let buffer = gethostname().map_err(std::io::Error::from)?.into_vec();
70        String::from_utf8(buffer).map_err(Error::InvalidUtf8String)
71    }
72
73    /// Get the architecture.
74    pub fn get_architecture(&self) -> Result<Option<Arch>, Error> {
75        let utsname = self.get_utsname()?;
76
77        // See:
78        // * https://en.wikipedia.org/wiki/Uname
79        // * https://stackoverflow.com/questions/45125516/possible-values-for-uname-m
80        match utsname.machine().to_str().ok_or(Error::InvalidUtf8OsStr)? {
81            "x86" | "i386" | "i586" | "i686" => Ok(Some(Arch::X86)),
82            "x86_64" | "amd64" => Ok(Some(Arch::X86_64)),
83            "arm" | "armv6l" | "armv7" | "armv7l" => Ok(Some(Arch::Arm)),
84            "aarch64_be" | "aarch64" | "armv8b" | "armv8l" => Ok(Some(Arch::AArch64)),
85            "ppc" => Ok(Some(Arch::PowerPc)),
86            "ppc64" | "ppc64le" => Ok(Some(Arch::PowerPc64)),
87            "sparc64" => Ok(Some(Arch::Sparc64)),
88            _ => Ok(None),
89        }
90    }
91
92    /// Get the system name.
93    pub fn get_system_name(&self) -> Result<Option<String>, Error> {
94        let utsname = self.get_utsname()?;
95        let name = utsname.sysname().to_str().ok_or(Error::InvalidUtf8OsStr)?;
96        let release = utsname.release().to_str().ok_or(Error::InvalidUtf8OsStr)?;
97
98        Ok(Some(format!("{name} {release}")))
99    }
100
101    /// Get the system version.
102    pub fn get_system_version(&self) -> Result<String, Error> {
103        let utsname = self.get_utsname()?;
104        let version = utsname.version().to_str().ok_or(Error::InvalidUtf8OsStr)?;
105
106        Ok(version.to_string())
107    }
108
109    /// Get the total amount of memory in the computer, in bytes
110    pub fn get_total_memory(&self) -> Result<u64, Error> {
111        let sysinfo = self.get_sysinfo()?;
112        Ok(sysinfo.ram_total())
113    }
114
115    /// Get the available amount of memory in the computer, in bytes
116    pub fn get_available_memory(&self) -> Result<u64, Error> {
117        let sysinfo = self.get_sysinfo()?;
118        Ok(sysinfo.ram_unused())
119    }
120
121    /// Get the total amount of swap in the computer, in bytes
122    pub fn get_total_swap(&self) -> Result<u64, Error> {
123        let sysinfo = self.get_sysinfo()?;
124        Ok(sysinfo.swap_total())
125    }
126
127    /// Get the available amount of swap in the computer, in bytes
128    pub fn get_available_swap(&self) -> Result<u64, Error> {
129        let sysinfo = self.get_sysinfo()?;
130        Ok(sysinfo.swap_free())
131    }
132
133    /// Get the number of logical cpu cores.
134    pub fn count_logical_cpus(&self) -> Result<usize, Error> {
135        Ok(sysconf(SysconfVar::_NPROCESSORS_ONLN)
136            .map_err(|error| Error::Io(std::io::Error::from(error)))?
137            .ok_or(Error::MissingValue)?
138            .try_into()
139            .expect("the number of cores cannot fit in a `usize`"))
140    }
141}
