30    /// Get a random list of catgirls.
31    ///
32    /// count is a num from 0 < count <= 100 and is the number of returned images.
33    /// nsfw is whether the images should be nsfw. If not specified, both are returned.
34    pub async fn get_random(&self, nsfw: Option<bool>, count: u8) -> Result<ImageList, Error> {
35        let mut buf = itoa::Buffer::new();
36        let count_query = std::iter::once(("count", buf.format(count.min(100))));
37        let nsfw_query = nsfw.map(|nsfw| ("nsfw", if nsfw { "true" } else { "false" }));
38        let query = count_query.chain(nsfw_query);
39        let url = Url::parse_with_params("https://nekos.moe/api/v1/random/image", query)?;
40
41        Ok(self
42            .client
43            .get(url.as_str())
44            .send()
45            .await?
46            .error_for_status()?
47            .json()
48            .await?)
49    }
50}
